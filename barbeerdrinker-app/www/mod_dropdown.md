`select:`
~~~sql
select column_names
from table_names
~~~

`insert:`
~~~sql
insert into table_name (column1, column2, ..., column_n)
values (value1, value2, ..., value_n)
~~~

`update:`
~~~sql
update table_name
set column1 = value1,
    column2 = value2,
    ...,
    column_n = value_n
where condition
~~~

`delete:`
~~~sql
delete
from table_name
where condition
~~~
